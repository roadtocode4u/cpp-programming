**💻 Example 1 :**
```cpp
#include <iostream>
using namespace std;

int main() {

    int l1=5, b2=8;

    int p1 = 2 * (l1+b1);

    cout<<p1;

     return 0;
}
```
**⚙️ Output :**
>26

**💻 Example 2 :**
```cpp
#include <iostream>
using namespace std;

int main() {

    int l1=5, b1=8;
    int p1 = 2 * (l1+b1);
    cout<<"Rectangle 1:"<<p1<<endl;

    int l2=5, b2=4;
    int p2 = 2 * (l2+b2);
    cout<<"Rectangle 2:"<<p2<<endl;

     return 0;
}
```
**⚙️ Output :**
>Rectangle 1: 26<br>
Rectangle 2: 18
